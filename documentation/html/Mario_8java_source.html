236  private TextureRegion getFrame(float dt) {
237  currentState = getState();
238 
239  TextureRegion region;
240 
241  switch (currentState) {
242  case DEAD:
243  region = marioDead;
244  break;
245  case JUMPING:
246  region = marioIsBig ? bigMarioJump : marioJump;
247  break;
248  case RUNNING:
249  region = marioIsBig ? bigMarioRun.getKeyFrame(stateTimer, true) : marioRun.getKeyFrame(stateTimer, true);
250  break;
251  case FALLING:
252  case STANDING:
253  default:
254  region = marioIsBig ? bigMarioStand : marioStand;
255  break;
256  }
257 
258  if ((body.getLinearVelocity().x < 0 || !runningRight) && !region.isFlipX()) {
259  region.flip(true, false);
260  runningRight = false;
261  }
262 
263  if ((body.getLinearVelocity().x > 0 || runningRight) && region.isFlipX()) {
264  region.flip(true, false);
265  runningRight = true;
266  }
267 
268  stateTimer = currentState == previousState ? stateTimer + dt : 0;
269  previousState = currentState;
270  return region;
271  }
